kquant API document (0.3.6)

kquant.chart

kquant.chart 모듈은 챠트 시각화 기능을 제공합니다.

사용법은 다음과 같습니다.

import kquant as kq
df = kq.daily_stock("005930", "2023-01-01", "2023-06-30")
kq.chart_line(df)
kq.chart_candle(df)
df = kq.daily_stock("005930", "2023-01-01", "2023-06-30")
df = kq.sma(df, 5, out_column="FAST")
df = kq.sma(df, 10, out_column="SLOW")
df = kq.cross(df, "FAST", "SLOW")
df["CROSS_UP_VALUE"] = None
df["CROSS_UP_VALUE"].loc[df["CROSS_UP"]] = df["FAST"]
kq.chart_line(df,
   {
       "FAST": {
           "line_color": "red",
       },
       "SLOW": {
           "line_color": "blue",
       },
       "CROSS_UP_VALUE": {
           "mode": "markers",
           "marker_color": "black",
           "marker_symbol": "triangle-up",
           "marker_size": 20,
       },
   },
   line_color="gray",
   line_dash="dot",
)
